# Set I2C bus to 0
i2c dev 0

# Read 6 bytes from EEPROM starting at address 0x9A
i2c read 0x58 0x9a 0 6 eeprom_mac

# Convert the binary data to a MAC address string
setexpr ethaddr fmt "%02x:%02x:%02x:%02x:%02x:%02x" \
    ${eeprom_mac:0:1} ${eeprom_mac:1:1} ${eeprom_mac:2:1} \
    ${eeprom_mac:3:1} ${eeprom_mac:4:1} ${eeprom_mac:5:1}

# Optionally save the environment with the new MAC address
saveenv
